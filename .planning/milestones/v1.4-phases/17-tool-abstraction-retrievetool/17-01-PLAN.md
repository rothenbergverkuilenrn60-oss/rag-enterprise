   class ToolRegistry:
       """Static class registry. Tools register via @get_tool_registry().register decorator.

       Provider-shape mapping for the LLM tool-schema wire format lives in
       schemas_for() — Anthropic uses ``input_schema``; OpenAI/Ollama wrap in
       ``{"type": "function", "function": {...}}``.
       """

       def __init__(self) -> None:
           self._tools: dict[str, type[BaseTool]] = {}

       def register(self, cls: type[BaseTool]) -> type[BaseTool]:
           """Register a Tool class. Returns cls unchanged so @decorator works."""
           if cls.name in self._tools:
               raise ValueError(f"Tool {cls.name!r} already registered")
           self._tools[cls.name] = cls
           return cls

       def get(self, name: str) -> BaseTool:
           """Instantiate a fresh tool per call (stateless dispatch)."""
           try:
               return self._tools[name]()
           except KeyError:
               raise KeyError(f"No tool registered as {name!r}") from None

       def list(self) -> list[str]:
           """Sorted names of all registered tools."""
           return sorted(self._tools.keys())

           self,
           provider: str,
           names: list[str] | None = None,
       ) -> list[dict[str, Any]]:
           """Provider-shape mapping. Anthropic vs OpenAI/Ollama wire formats."""
           tools = [
               t for t in self._tools.values()
               if names is None or t.name in names
           ]
           if provider == "anthropic":

                       "description":  t.description,
                       "input_schema": t.parameters_schema,
                   }
                   for t in tools
               ]
           if provider in ("openai", "ollama"):
               return [
                   {
                       "type": "function",
                       "function": {
                           "name":        t.name,
                           "description": t.description,
                           "parameters":  t.parameters_schema,
                       },
                   }
                   for t in tools
               ]
           raise ValueError(f"Unknown provider: {provider!r}")
