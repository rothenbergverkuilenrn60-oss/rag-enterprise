@get_tool_registry().register
class WebSearchTool(BaseTool):
    """web_search ‚ÄĒ real-time/external retrieval via Tavily."""

    name: ClassVar[str] = "web_search"
    description: ClassVar[str] = (
        "Search the public web for current/real-time information, news, "
        "recent events, or topics not covered by the internal knowledge "
        "base. Prefer search_knowledge_base for indexed corpus questions."
    )
    parameters_schema: ClassVar[dict[str, Any]] = _WEB_SEARCH_PARAMETERS_SCHEMA

    async def run(self, args: dict[str, Any], ctx: ToolContext) -> ToolResult:
        t0 = time.perf_counter()
        a = args or {}
        query_str = (a.get("query") or ctx.req.query or "").strip()

        # 1. tavily_disabled short-circuit (D-03) ‚ÄĒ BEFORE retry boundary,
        #    BEFORE any network call, BEFORE get_tavily_client invocation.
        if not settings.tavily_api_key:
            return ToolResult(
                content="Web search not configured. Answer from the knowledge base only.",
                chunks=[],
                metadata={"error": True, "kind": "tavily_disabled", "latency_ms": 0},
                is_error=True,
            )

        # 2. Real Tavily call via tenacity-wrapped helper.
        try:
            resp = await _tavily_search(query_str)
        except httpx.HTTPStatusError as exc:
            latency_ms = int((time.perf_counter() - t0) * 1000)
            status = exc.response.status_code if exc.response is not None else 0

            )

                chunks=[],
                metadata={"error": True, "kind": "web_search_failed", "latency_ms": latency_ms},
                is_error=True,
            )

            chunk_id = f"web:{hashlib.sha1(url.encode('utf-8')).hexdigest()[:16]}"
            chunks.append(
                RetrievedChunk(
                    chunk_id=chunk_id,
                    doc_id="web",
                    content=r.get("content", ""),
                    metadata=ChunkMetadata(
                        source=url,
                        title=r.get("title", ""),
                        chunk_type="web",
                        page_number=None,
                    ),
                    final_score=float(r.get("score", 0.0)),
                    retrieval_method="web",
                )
            )

        latency_ms = int((time.perf_counter() - t0) * 1000)
        # `content` text ‚ÄĒ concise human-readable summary; NEVER includes URLs/snippets
        # raw beyond what the chunks already carry. Planner sees both `content` and `chunks`.
        ctx_text = f"Web search returned {len(chunks)} result(s)."
        return ToolResult(
            content=ctx_text,
            chunks=chunks,
            metadata={
                "latency_ms": latency_ms,
                "query": query_str,
                "chunk_count": len(chunks),
            },
        )
